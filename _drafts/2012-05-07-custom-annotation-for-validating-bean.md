validate(Object obj)<br /> {<br />  List&lt;String&gt; errors = new ArrayList&lt;String&gt;();<br /><br />  NotNullAndNotEmpty annotations = (NotNullAndNotEmpty) obj.getClass()<br />    .getAnnotation(NotNullAndNotEmpty.class);<br />  <br />  //if annotation not found on class then return empty list<br />  if (annotations != null)<br />  {<br /><br />   Field[] fields = obj.getClass().getDeclaredFields();<br />   for (Field field : fields)<br />   {<br />    field.setAccessible(true);<br />    try<br />    {<br />     if (field.get(obj) == null)<br />     {<br />      errors.add(field.getName() + " is null or empty");<br />     }<br />    }<br />    catch (Exception e)<br />    {<br />     e.printStackTrace();<br />    }<br />   }<br />  }<br />  return errors;<br /> }<br />}<br /></pre> <u><b>Usage :- Model</b></u><br /> <pre class="brush: java"><br />package com;<br /><br />@NotNullAndNotEmpty<br />public class Employee<br />{<br /> private String name;<br /> private String address;<br /><br /> public Employee()<br /> {<br /><br /> }<br /><br /> public Employee(String name, String address)<br /> {<br />  super();<br />  this.name = name;<br />  this.address = address;<br /> }<br /><br /> public String getName()<br /> {<br />  return name;<br /> }<br /><br /> public void setName(String name)<br /> {<br />  this.name = name;<br /> }<br /><br /> public String getAddress()<br /> {<br />  return address;<br /> }<br /><br /> public void setAddress(String address)<br /> {<br />  this.address = address;<br /> }<br /><br />}<br /></pre> <u><b>Run the test</b></u><br /> <pre class="brush: java"><br />package com;<br /><br />import java.util.List;<br /><br />public class Test<br />{<br /><br /> public static void main(String[] args)<br /> {<br />  Employee emp = new Employee("Sandeep", null);<br /><br />  List&lt;String&gt;
